U172713 01多源最短路

代码如下

#include<bits/stdc++.h>
#define N 1001

using namespace std;

inline int read() {
    int w=1,x=0;
    char ch=0;
    while(ch<'0'||ch>'9') {if(ch=='-') w=-1;ch=getchar();}
    while(ch>='0'&&ch<='9') x=x*10+ch-'0',ch=getchar();
    return w*x;
}

int n;
bitset<N> g[N],vis,now;
int dis[N][N];

int main() {
    n=read();
    for(int i=1;i<=n;i++) {
        for(int j=1;j<=n;j++) {
            int x=read();
            if(x) g[i].set(j);
        }
    }
    memset(dis,-1,sizeof(dis));
    for(int i=1;i<=n;i++) {
        vis.reset();
        vis.set(i);
        queue<int> q;
        q.push(i);
        dis[i][i]=0;
        while(!q.empty()) {
            int u=q.front();q.pop();
            now=g[u]^(g[u]&vis);
            for(int v=now._Find_first();v!=now.size();v=now._Find_next(v)) {
                dis[i][v]=dis[i][u]+1;
                q.push(v);
            }
            vis|=now;
        }
    }
    for(int i=1;i<=n;i++) {
        for(int j=1;j<=n;j++) printf("%d ",dis[i][j]);
        printf("\n");
    }
    system("pause");
    return 0;
}
