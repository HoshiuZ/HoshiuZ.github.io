洛谷5463 小鱼比可爱（加强版）

思路

考虑一个逆序对 $(i,j)$ 的贡献，显然为 $i\times (n-j+1)$ 。

所以可以考虑维护一个权值线段树，每次加下标。

单点修改，求后缀和。

本题会爆 long long ，啥时候去学学 _int128 。

代码

#include<bits/stdc++.h>
#define ll long long 
#define N 1000010
using namespace std;

int n;
ll a[N];

class SegmentTree{
  private:
    ll sum[N<<2];
    #define s(x) sum[x]
    #define ls (p<<1)
    #define rs (ls|1)
    #define mid (l+r>>1)
    void update(int p) {
        s(p)=s(ls)+s(rs);
    }
  public:
    void build(int p,int l,int r) {
        if(l==r) {
            s(p)=0;
            return ;
        }
        build(ls,l,mid),build(rs,mid+1,r);
        update(p);
    }
    void change(int p,int l,int r,int x,int y) {
        if(l>x||r<x) return ;
        if(l==r) {
            s(p)+=y;
            return ;
        }
        change(ls,l,mid,x,y),change(rs,mid+1,r,x,y);
        update(p);
    }
    int ask(int p,int l,int r,int L,int R) {
        if(l>R||r<L) return 0;
        if(l>=L&&r<=R) return s(p);
        return ask(ls,l,mid,L,R)+ask(rs,mid+1,r,L,R);
    }
}SMT;

int raw[N],t;
map<ll,int> val;
vector<ll> tmp;
void discrete() {
    sort(tmp.begin(),tmp.end());
    t=unique(tmp.begin(),tmp.end())-tmp.begin();
    for(int i=0;i<t;i++) {
        val[tmp[i]]=i+1;
        raw[i+1]=tmp[i];
    }
}

void solve() {
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%lld",&a[i]);
    for(int i=1;i<=n;i++) tmp.push_back(a[i]);
    discrete();
    SMT.build(1,1,t);
    ll ans=0;
    for(int i=1;i<=n;i++) {
        ans+=(ll)(n-i+1)*SMT.ask(1,1,t,val[a[i]]+1,t);
        SMT.change(1,1,t,val[a[i]],i);
    }
    printf("%lld\n",ans);
}

int main() {
    solve();
    return 0;
}

JLOI2011 不等式组

思路

不等式组是不能维护的，所以需要转化问题。

分类讨论 $a$ 的取值。

  • 若 $a=0$ ，那么如果 $b>c$ 的话，不等式恒成立，否则恒不成立。
  • 若 $a>0$ ，那么可以化为 $x>\dfrac{c-b}{a}$ ，而均为整数，所以 $x\ge \Big\lfloor\dfrac{c-b}{a}\Big\rfloor+1$ 。
  • 若 $a<0$ ，那么可以化为 $x<\dfrac{c-b}{a}$ ，而均为整数，所以 $x\le \Big\lfloor\dfrac{c-b}{a}\Big\rfloor-1$ 。

于是可以考虑建两棵线段树，一个维护大于号，一个维护小于号。

对于操作一，分类讨论，视情况判断是否要在线段树上单点修改。

对于操作二，同样的分类讨论，视情况判断是否要在线段树上单点修改。

对于操作三，两棵线段树区间查询，加上恒成立的个数即可。

细节较多，需要处理一大堆的特殊情况。

因为数组开小了调了半天。（ $10^6$ 开成 $10^5$ ）

可能为负，所以需要值域平移。

代码

#include<bits/stdc++.h>
#define N 1000010

using namespace std;

int n;
int tot=0,val[N];
bool tp[N],flag[N],aez[N];
const int M=1000000;

class SegmentTree{
  private:
    int sum[N<<3];
    #define s(x) sum[x]
    #define ls (p<<1)
    #define rs (ls|1)
    #define mid (l+r>>1)
    void update(int p) {
        s(p)=s(ls)+s(rs);
    }
  public:
    void build(int p,int l,int r) {
        if(l==r) {
            s(p)=0;
            return ;
        }
        build(ls,l,mid),build(rs,mid+1,r);
        update(p);
    }
    void change(int p,int l,int r,int x,int y) {
        if(l>x||r<x) return ;
        if(l==r) {
            s(p)+=y;
            return ;
        }
        change(ls,l,mid,x,y),change(rs,mid+1,r,x,y);
        update(p);
    }
    int ask(int p,int l,int r,int L,int R) {
        if(l>R||r<L) return 0;
        if(l>=L&&r<=R) return s(p);
        return ask(ls,l,mid,L,R)+ask(rs,mid+1,r,L,R);
    }
}TL,TR;

void solve() {
    scanf("%d",&n);
    TL.build(1,1,2*M+1),TR.build(1,1,2*M+1);
    int h=0;
    for(int i=1;i<=n;i++) {
        string op;
        cin>>op;
        if(op[0]=='A') {
            int a,b,c;
            scanf("%d%d%d",&a,&b,&c);
            if(!a) {
                if(b>c) h++,val[++tot]=0;
                else val[++tot]=-1;
                aez[tot]=true;
            }
            else {
                int y;
                if(a>0) {
                    y=floor((c*1.0-b)/a)+1+M+1;
                    val[++tot]=y;
                    if(y<1) h++;
                    else if(y<=2*M+1) TL.change(1,1,2*M+1,y,1);
                }
                else {
                    y=ceil((c*1.0-b)/a)-1+M+1;
                    val[++tot]=y;
                    tp[tot]=true;
                    if(y>2*M+1) h++;
                    else if(y>=1) TR.change(1,1,2*M+1,y,1);
                }
            }
        }
        else if(op[0]=='D') {
            int x;
            scanf("%d",&x);
            if(flag[x]) continue;
            flag[x]=true;
            if(aez[x]) {
                if(val[x]==0) h--;
                continue;
            }
            if(val[x]>2*M+1||val[x]<1) {
                if(val[x]>2*M+1&&tp[x]) h--;
                if(val[x]<1&&!tp[x]) h--;
            }
            else {
                if(tp[x]) TR.change(1,1,2*M+1,val[x],-1);
                else TL.change(1,1,2*M+1,val[x],-1);
            }
        }
        else {
            int k;
            scanf("%d",&k);
            int ans=0;
            ans+=h+TL.ask(1,1,2*M+1,1,k+M+1)+TR.ask(1,1,2*M+1,k+M+1,2*M+1);
            printf("%d\n",ans);
        }
    }
}

int main() {
    solve();
    system("pause");
    return 0;
}

洛谷1637 三元上升子序列

思路

类似于逆序对的权值线段树/树状数组思想。

比较典。

#include<bits/stdc++.h>

    }nod[N<<2];

    #define ls (p<<1)
    #define rs (ls|1)

        s(p)=s(ls)+s(rs);
    }
  public:

    scanf("%d",&n);
    int m=0;

        SSMT.change(1,1,m,a[i],1);
    }
    printf("%lld\n",ans);
}

    work();
    return 0;
}

洛谷T22526 Princess Principal

思路

考虑一个贪心，分段时，对于一个段的起点 $l$ ，其终点 $r$ 是满足 $s(l,r)\le k,s(l,r+1)>k$ 。

于是对每个 $i$ ，利用双指针 $O(n)$ 求出 $f[i]$ 表示以 $i$ 为起点的段，终点的后一个位置。

所以在处理一个区间查询时，可以从 $l$ 开始跳，即循环 $l=f[l]$ ，直到 $l>r$ 为止。

但是一个一个跳的话如果遇到极端情况比如全是 $1$ 且 $k=1$ ，那就很容易被卡掉。

所以可以考虑倍增，即预处理出 $f[n][i]$ 表示从位置 $n$ 跳 $2^i$ 次到达的位置。

于是对每个询问即可 $O(\log n)$ 的求出了。

时间复杂度 $O(n\log n)$ 。

代码

#include<bits/stdc++.h>
#define ll long long 
#define N 1000010
#define Log 21

using namespace std;

int n,m,k;
ll a[N],sum[N],f[N][Log];

void work() {
    scanf("%d%d%d",&n,&m,&k);
    for(int i=1;i<=n;i++) scanf("%lld",&a[i]);
    for(int i=1;i<=n;i++) sum[i]=sum[i-1]+a[i];
    int r=1;
    for(int l=1;l<=n;l++) {
        while(sum[r]-sum[l-1]<=k&&r<=n) r++;
        f[l][0]=r;
        if(a[l]>k) r++;
    }
    f[n+1][0]=n+1;
    for(int j=1;j<Log;j++) {
        for(int i=1;i<=n;i++) f[i][j]=f[f[i][j-1]][j-1];
        f[n+1][j]=n+1;
    }
    for(int i=1;i<=m;i++) {
        int L,R,ans=1,s=0;
        scanf("%d%d",&L,&R);
        if(f[L][Log-1]<=R) printf("Chtholly\n");
        else {
            for(int j=Log-1;j>=0;j--) {
                if(f[L][j]<=R) L=f[L][j],ans+=1<<j;
            }
            printf("%d\n",ans);
        }
    }
}

int main() {
    work();
    return 0;
}

洛谷4513 小白逛公园

思路

以前刚学线段树的时候做过这题。

考虑用线段树维护区间和 $sum$ ，左最大子段和 $lmaxn$ ，右最大子段和 $rmaxn$ ，最大子段和 $maxn$ 。

考虑如何合并区间。

$sum$ 不用解释。

$lmaxn$ 有两种可能，一种是左子节点的 $lmaxn$ ，另一种是左子节点的 $sum$ 加上右子节点的 $lmaxn$ ，两者取 $\max$ 。

$rmaxn$ 其实同理。

$maxn$ 有三种可能，一种是左子节点的 $maxn$ ，一种是右子节点的 $maxn$ ，一种是左子节点的 $rmaxn$ 加上右子节点的 $lmaxn$ 。

于是便可以用线段树 $O(n\log n)$ 解决本题。

代码

#include<bits/stdc++.h>
#define N 500010

using namespace std;

int n,m,a[N];

class SegmentTree{
  private:
    struct node{
        int sum,lmaxn,rmaxn,maxn;
    }nod[N<<2];
    #define s(p) nod[p].sum
    #define lm(p) nod[p].lmaxn
    #define rm(p) nod[p].rmaxn
    #define m(p) nod[p].maxn
    #define mid (l+r>>1)
    #define ls (p<<1)
    #define rs (ls|1)
    void update(int p) {
        s(p)=s(ls)+s(rs);
        m(p)=max(rm(ls)+lm(rs),max(m(ls),m(rs)));
        lm(p)=max(lm(ls),s(ls)+lm(rs));
        rm(p)=max(rm(rs),s(rs)+rm(ls));
    }
  public:
    void change(int p,int l,int r,int x,int y) {
        if(l>x||r<x) return ;
        if(l==r) {
            s(p)=m(p)=lm(p)=rm(p)=y;
            return ;
        }
        change(ls,l,mid,x,y),change(rs,mid+1,r,x,y);
        update(p);
    }
    node query(int p,int l,int r,int L,int R) {
        if(l>=L&&r<=R) return nod[p];
        if(R<=mid) return query(ls,l,mid,L,R);
        if(L>=mid+1) return query(rs,mid+1,r,L,R);
        node ln=query(ls,l,mid,L,R),rn=query(rs,mid+1,r,L,R),nd;
        nd.sum=ln.sum+rn.sum;
        nd.lmaxn=max(ln.lmaxn,ln.sum+rn.lmaxn);
        nd.rmaxn=max(rn.rmaxn,rn.sum+ln.rmaxn);
        nd.maxn=max(ln.rmaxn+rn.lmaxn,max(ln.maxn,rn.maxn));
        return nd;
    }
    int ask(int p,int l,int r,int L,int R) {
        node ans=query(p,l,r,L,R);
        return ans.maxn;
    }
}SMT;

void work() {
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    for(int i=1;i<=n;i++) SMT.change(1,1,n,i,a[i]);
    for(int i=1;i<=m;i++) {
        int k,l,r,p,s;
        scanf("%d",&k);
        if(k==1) {
            scanf("%d%d",&l,&r);
            if(l>r) swap(l,r);
            printf("%d\n",SMT.ask(1,1,n,l,r));
        }
        else {
            scanf("%d%d",&p,&s);
            SMT.change(1,1,n,p,s);
        }
    }
}

int main() {
    work();
    return 0;
}

CDQZ challenge 4

给一个长为N的数列，有M次操作，每次操作时以下三种之一：

（1）修改数列中的一个数

（2）求数列中某连续一段所有数的两两乘积的和 mod 1000000007

（3）求数列中某连续一段所有相邻两数乘积的和 mod 1000000007

思路

不算难，比较好想的一题。

考虑如何维护两两乘积。

对于节点 $x$ ，有子节点 $y,z$ ，那么 $x$ 的两两乘积等于 $y$ 的两两乘积加 $z$ 的两两乘积加 $y$ 的和乘 $z$ 的和。

考虑如何为维护相邻两数乘积。

对于节点 $x$，有子节点 $y,z$ ，那么 $x$ 的相邻两数乘积等于 $y$ 的相邻两数乘积加 $z$ 的相邻两数乘积加 $y$ 的最后一个数乘 $z$ 的第一个数（假设 $y$ 为左节点， $z$ 为右节点）。

于是直接用线段树维护即可。

代码

#include<bits/stdc++.h>
#define ll long long 
#define N 100010
#define mod 1000000007

using namespace std;

int n,m;
ll a[N];

class SegmentTree{
  private:
    struct node{
        ll sum,msum,val,lval,rval;
    }nod[N<<2];
    #define s(p) nod[p].sum
    #define ms(p) nod[p].msum
    #define v(p) nod[p].val
    #define lv(p) nod[p].lval
    #define rv(p) nod[p].rval
    #define mid (l+r>>1)
    #define ls (p<<1)
    #define rs (ls|1)
    void update(int p) {
        (s(p)=s(ls)+s(rs))%=mod;
        (ms(p)=ms(ls)+ms(rs)+s(ls)*s(rs))%=mod;
        (v(p)=v(ls)+v(rs)+rv(ls)*lv(rs))%=mod;
        lv(p)=lv(ls),rv(p)=rv(rs);
    }
  public:
    void change(int p,int l,int r,int x,ll y) {
        if(l>x||r<x) return ;
        if(l==r) {
            s(p)=lv(p)=rv(p)=y%mod;
            ms(p)=v(p)=0;
            return ;
        }
        change(ls,l,mid,x,y),change(rs,mid+1,r,x,y);
        update(p);
    }
    node query(int p,int l,int r,int L,int R) {
        if(l>=L&&r<=R) return nod[p];
        if(R<=mid) return query(ls,l,mid,L,R);
        if(L>=mid+1) return query(rs,mid+1,r,L,R);
        node ln=query(ls,l,mid,L,R),rn=query(rs,mid+1,r,L,R),ans;
        (ans.sum=ln.sum+rn.sum)%=mod;
        (ans.msum=ln.msum+rn.msum+ln.sum*rn.sum)%=mod;
        (ans.val=ln.val+rn.val+ln.rval*rn.lval)%=mod;
        ans.lval=ln.lval,ans.rval=rn.rval;
        return ans;
    }
    ll ask_msum(int p,int l,int r,int L,int R) {return (query(p,l,r,L,R).msum+mod)%mod;}
    ll ask_val(int p,int l,int r,int L,int R) {return (query(p,l,r,L,R).val+mod)%mod;}
}SMT;

void work() {
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%lld",&a[i]);
    for(int i=1;i<=n;i++) SMT.change(1,1,n,i,a[i]);
    for(int i=1;i<=m;i++) {
        char op;
        int l,r,x;
        ll y;
        scanf(" %c",&op);
        if(op=='M') {
            scanf("%d%lld",&x,&y);
            SMT.change(1,1,n,x,y);
        }
        else if(op=='Q') {
            scanf("%d%d",&l,&r);
            printf("%lld\n",SMT.ask_msum(1,1,n,l,r));
        }
        else {
            scanf("%d%d",&l,&r);
            printf("%lld\n",SMT.ask_val(1,1,n,l,r));
        }
    }
}

int main() {
    work();
    return 0;
}

洛谷7706 「Wdsr-2.7」文文的摄影布置

思路

考虑如何维护这个 $\psi$ 。

对于区间 $x$ ，设其子节点 $y$ 和 $z$ ，那么 $\psi_z$ 应该等于 $\psi_y$ 和 $\psi_z$ 和 $AB_y+A_z$ 和 $A_y+BA_z$ 之间的较大值，其中 $AB$ 为区间内 $A_i-B_j$ 的最大值，$i<j$ ；$BA$ 为区间内 $A_k-B_j$ 的最大值，$j<k$ 。 $A$ 为区间内 $A$ 的最大值， $B$ 为区间内 $B$ 的最小值。

那么只需要用线段树维护 $\psi$ ，$AB$ ， $BA$ ，$A$， $B$ 即可。

代码

 #include<bits/stdc++.h>
#define N 500010
#define INF 0x3f3f3f3f

using namespace std;

int n,m,a[N],b[N];

class SegmentTree{
  private:
    struct node{
        int maxa,minb,maxab,maxba,maxphi;
    }nod[N<<2];
    #define ma(p) nod[p].maxa
    #define mb(p) nod[p].minb
    #define mab(p) nod[p].maxab
    #define mba(p) nod[p].maxba
    #define mp(p) nod[p].maxphi
    #define mid (l+r>>1)
    #define ls (p<<1)
    #define rs (ls|1)
    void update(int p) {
        ma(p)=max(ma(ls),ma(rs));
        mb(p)=min(mb(ls),mb(rs));
        mab(p)=max(ma(ls)-mb(rs),max(mab(ls),mab(rs)));
        mba(p)=max(ma(rs)-mb(ls),max(mba(ls),mba(rs)));
        mp(p)=max(max(mp(ls),mp(rs)),max(mab(ls)+ma(rs),ma(ls)+mba(rs)));
    }
  public:
    void build(int p,int l,int r) {
        if(l==r) {
            ma(p)=a[l];
            mb(p)=b[l];
            mab(p)=mba(p)=mp(p)=-INF;
            return ;
        }
        build(ls,l,mid),build(rs,mid+1,r);
        update(p);
    }
    void change_a(int p,int l,int r,int x,int y) {
        if(l>x||r<x) return ;
        if(l==r) {
            ma(p)=y;
            return ;
        }
        change_a(ls,l,mid,x,y),change_a(rs,mid+1,r,x,y);
        update(p);
    }
    void change_b(int p,int l,int r,int x,int y) {
        if(l>x||r<x) return ;
        if(l==r) {
            mb(p)=y;
            return ;
        }
        change_b(ls,l,mid,x,y),change_b(rs,mid+1,r,x,y);
        update(p);
    }
    node query(int p,int l,int r,int L,int R) {
        if(l>=L&&r<=R) return nod[p];
        if(R<=mid) return query(ls,l,mid,L,R);
        if(L>=mid+1) return query(rs,mid+1,r,L,R);
        node ln=query(ls,l,mid,L,R),rn=query(rs,mid+1,r,L,R),ans;
        ans.maxa=max(ln.maxa,rn.maxa);
        ans.minb=min(ln.minb,rn.minb);
        ans.maxab=max(ln.maxa-rn.minb,max(ln.maxab,rn.maxab));
        ans.maxba=max(rn.maxa-ln.minb,max(ln.maxba,rn.maxba));
        ans.maxphi=max(max(ln.maxphi,rn.maxphi),max(ln.maxab+rn.maxa,ln.maxa+rn.maxba));
        return ans;
    }
    int ask(int p,int l,int r,int L,int R) {
        return query(p,l,r,L,R).maxphi;
    }
}SMT;

void work() {
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    for(int i=1;i<=n;i++) scanf("%d",&b[i]);
    SMT.build(1,1,n);
    for(int i=1;i<=m;i++) {
        int op,x,y;
        scanf("%d%d%d",&op,&x,&y);
        if(op==1) SMT.change_a(1,1,n,x,y);
        else if(op==2) SMT.change_b(1,1,n,x,y);
        else printf("%d\n",SMT.ask(1,1,n,x,y));
    }
}

int main() {
    work();
    return 0;
}

洛谷6033 [NOIP2004 提高组] 合并果子 加强版

思路

直接贴吧。

代码

#include<bits/stdc++.h>
#define N 10000010
#define M 100010
#define ll long long 

using namespace std;

inline ll read() {
	ll w=1,x=0;
	char ch=0;
	while(ch<'0'||ch>'9') {if(ch=='-') w=-1;ch=getchar();}
	while(ch>='0'&&ch<='9') x=x*10+ch-'0',ch=getchar();
	return w*x;
}

int n,a[N],cnt[M];

ll ans=0;
queue<ll> aq,q;

ll get_min(queue<ll> &x,queue<ll> &y) {
    ll res=0;
    if(x.empty()) res=y.front(),y.pop();
    else if(y.empty()) res=x.front(),x.pop();
    else x.front()<y.front()?(res=x.front(),x.pop()):(res=y.front(),y.pop());
    return res;
}

void work() {
    n=read();
    for(int i=1;i<=n;i++) {a[i]=read();cnt[a[i]]++;}
    for(int i=1;i<M;i++) while(cnt[i]--) aq.push(i);
    while(aq.size()+q.size()>1) {
        ll x=0,y=0;
        x=get_min(q,aq);
        y=get_min(q,aq);
        ans+=x+y;
        q.push(x+y);
    }
    printf("%lld\n",ans);
}

int main() {
    work();
    return 0;
}

#include<bits/stdc++.h>
#define ll long long 
#define N 100010
#define mp(x,y) make_pair(x,y)

using namespace std;

int n,w;
ll a[N],sum[N];
struct node{
    int l,r;
    ll val;
    node(int _l,int _r,ll _val):l(_l),r(_r),val(_val){}
    bool operator < (const node &x) const {
        return val<x.val;
    }
};
priority_queue<node> q;
map<pair<int,int>,bool> vh;

void work() {
    scanf("%d%d",&n,&w);
    for(int i=1;i<=n;i++) scanf("%lld",&a[i]);
    for(int i=1;i<=n;i++) sum[i]=sum[i-1]+a[i];
    q.push(node(1,n,sum[n]));
    vh[mp(1,n)]=true;
    int cnt=0;
    while(cnt++<w) {
        node u=q.top();
        q.pop();
        printf("%lld ",u.val);
        if(u.l+1<=u.r&&!vh[mp(u.l+1,u.r)]) q.push(node(u.l+1,u.r,sum[u.r]-sum[u.l])),vh[mp(u.l+1,u.r)]=true;
        if(u.l<=u.r-1&&!vh[mp(u.l,u.r-1)]) q.push(node(u.l,u.r-1,sum[u.r-1]-sum[u.l-1])),vh[mp(u.l,u.r-1)]=true;
    }
    printf("\n");
}

[SHOI2015] 脑洞治疗仪

思路

直接贴吧。

•使用线段树维护序列，对每个节点维护其是否为全1，以及内部最长0段，以及左右最长0段

•对于3操作，我们查区间和，这样就可以知道区间有多少1了

•然后我们在填到的区间中二分一下填到哪个位置用完了我们挖出来的1

•为了实现这两步需要维护每个节点代表的区间和

•然后就把3操作转换为了：

•区间修改为0或1

•这样的操作

•每次直接二分位置然后线段树的话时间复杂度为O(n+mlog^2n)

•可以做到O(n+mlogn)，在线段树上二分即可

————上述来自 PPT 。

代码

#include<bits/stdc++.h>
#define N 200010

using namespace std;

class SegmentTree{
  private:
    struct node{
        int num,cnt,lcnt,rcnt,lazy;
    }nod[N<<2];
    #define n(p) nod[p].num
    #define c(p) nod[p].cnt
    #define lc(p) nod[p].lcnt
    #define rc(p) nod[p].rcnt
    #define la(p) nod[p].lazy
    #define ls (p<<1)
    #define rs (ls|1)
    #define mid (l+r>>1)
    void update(int p,int l,int r) {
        n(p)=n(ls)+n(rs);
        c(p)=max(rc(ls)+lc(rs),max(c(ls),c(rs)));
        lc(p)=lc(ls);if(c(ls)==mid-l+1) lc(p)=lc(ls)+lc(rs);
        rc(p)=rc(rs);if(c(rs)==r-mid) rc(p)=rc(rs)+rc(ls);
    }
    void spread(int p,int l,int r) {
        if(la(p)==-1) return ;
        la(ls)=la(rs)=la(p);
        if(!la(p)) {
            n(ls)=c(ls)=lc(ls)=rc(ls)=mid-l+1;
            n(rs)=c(rs)=lc(rs)=rc(rs)=r-mid;
        }
        else n(ls)=c(ls)=lc(ls)=rc(ls)=n(rs)=c(rs)=lc(rs)=rc(rs)=0;
        la(p)=-1;
    }
  public:
    void build(int p,int l,int r) {
        n(p)=c(p)=lc(p)=rc(p)=0;
        la(p)=-1;
        if(l==r) return ;
        build(ls,l,mid),build(rs,mid+1,r);
    }
    void change(int p,int l,int r,int L,int R,int x) {
        if(l>R||r<L) return ;
        if(l>=L&&r<=R) {
            if(!x) n(p)=c(p)=lc(p)=rc(p)=r-l+1;
            else n(p)=c(p)=lc(p)=rc(p)=0;
            la(p)=x;
            return ;
        }
        spread(p,l,r);
        change(ls,l,mid,L,R,x),change(rs,mid+1,r,L,R,x);
        update(p,l,r);
    }
    node query(int p,int l,int r,int L,int R) {
        if(l>=L&&r<=R) return nod[p];
        spread(p,l,r);
        if(R<=mid) return query(ls,l,mid,L,R);
        if(L>=mid+1) return query(rs,mid+1,r,L,R);
        node ln=query(ls,l,mid,L,R),rn=query(rs,mid+1,r,L,R),ans;
        ans.num=ln.num+rn.num;
        ans.cnt=max(ln.rcnt+rn.lcnt,max(ln.cnt,rn.cnt));
        ans.lazy=-1;
        ans.lcnt=ln.lcnt;if(ln.lcnt==mid-l+1) ans.lcnt=ln.lcnt+rn.lcnt;
        ans.rcnt=rn.rcnt;if(rn.rcnt==r-mid) ans.rcnt=rn.rcnt+ln.rcnt;
        return ans;
    }
    int ask_num(int p,int l,int r,int L,int R) {
        return query(p,l,r,L,R).num;
    }
    int ask_cnt(int p,int l,int r,int L,int R) {
        return query(p,l,r,L,R).cnt;
    }
}SMT;

int n,m;
void work() {   
    scanf("%d%d",&n,&m);
    SMT.build(1,1,n);
    for(int i=1;i<=m;i++) {
        int op,l,r,L,R;
        scanf("%d",&op);
        if(op==0) {
            scanf("%d%d",&l,&r);
            SMT.change(1,1,n,l,r,0);
        }
        else if(op==1) {
            scanf("%d%d%d%d",&l,&r,&L,&R);
            int s=r-l+1-SMT.ask_num(1,1,n,l,r); 
            SMT.change(1,1,n,l,r,0);
            if(!s) continue;
            int h=L,t=R+1;
            while(h+1<t) {
                int c=h+t>>1;
                if(SMT.ask_num(1,1,n,L,c)>s) t=c;
                else h=c;
            }
            SMT.change(1,1,n,L,h,1);
        }
        else {
            scanf("%d%d",&l,&r);
            printf("%d\n",SMT.ask_cnt(1,1,n,l,r));
        }
    }

}

int main() {
    work();
    return 0;
}

HDU 6315 Naive Operations

思路

代码

#include<bits/stdc++.h>
#define ll long long 
#define N 100010

using namespace std;

int n,m,b[N];

class SegmentTree{
  private:
    struct node{
        int maxn,lazy;
        ll sum;
    }nod[N<<2];
    #define m(p) nod[p].maxn
    #define la(p) nod[p].lazy
    #define s(p) nod[p].sum
    #define mid (l+r>>1)
    #define ls (p<<1)
    #define rs (ls|1)
    void update(int p) {
        m(p)=max(m(ls),m(rs));
        s(p)=s(ls)+s(rs);
    }
    void spread(int p,int l,int r) {
        if(!la(p)) return ;
        m(ls)+=la(p),la(ls)+=la(p);
        m(rs)+=la(p),la(rs)+=la(p);
        la(p)=0;
    }
  public:
    void build(int p,int l,int r) {
        la(p)=0;
        if(l==r) {
            m(p)=-b[l];
            s(p)=0;
            return ;
        }
        build(ls,l,mid),build(rs,mid+1,r);
        update(p);
    }
    void change(int p,int l,int r,int L,int R) {
        if(l>R||r<L) return ;
        if(l>=L&&r<=R) {
            m(p)++;
            la(p)++;
            if(m(p)==0) verify(p,l,r);
            return ;
        }
        spread(p,l,r);
        change(ls,l,mid,L,R),change(rs,mid+1,r,L,R);
        update(p);
    }
    void verify(int p,int l,int r) {
        if(m(p)<0) return ;
        if(l==r) {
            m(p)-=b[l];
            s(p)++;
            return ;
        }
        spread(p,l,r);
        verify(ls,l,mid),verify(rs,mid+1,r);
        update(p);
    }
    ll ask(int p,int l,int r,int L,int R) {
        if(l>R||r<L) return 0;
        if(l>=L&&r<=R) return s(p);
        return ask(ls,l,mid,L,R)+ask(rs,mid+1,r,L,R);
    }
}SMT;

void work() {
    while(scanf("%d%d",&n,&m)!=EOF){
        for(int i=1;i<=n;i++) scanf("%d",&b[i]);
        SMT.build(1,1,n);
        for(int i=1;i<=m;i++) {
            char op[10];
            int l,r;
            scanf("%s",op);
            scanf("%d%d",&l,&r);
            if(op[0]=='a') SMT.change(1,1,n,l,r);
            else printf("%lld\n",SMT.ask(1,1,n,l,r));
        }
    }
}

int main() {
    work();
    return 0;
}
